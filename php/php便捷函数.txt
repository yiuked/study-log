__autoload()  
自动载入class文件
function __autoload( $class ) {
 $file = $class . '.class.php';  
 if ( is_file($file) ) {  
  require_once($file);  
 }
}

spl_autoload_register()
当用户在找不到类的时候，则自动调用spl_autoload_register的参数函数
function loadprint( $class ) {
 $file = $class . '.class.php';  
 if (is_file($file)) {  
  require_once($file);  
 } 
}
spl_autoload_register( 'loadprint' );
